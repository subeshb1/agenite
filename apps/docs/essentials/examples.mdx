## Basic examples

### Simple chat agent

Create a basic chat agent with a calculator tool:

```typescript
import { Agent } from '@agenite/agent';
import { Tool } from '@agenite/tool';
import { OllamaProvider } from '@agenite/ollama';

const calculatorTool = new Tool({
  name: 'calculator',
  description: 'Perform basic math operations',
  version: '1.0.0',
  inputSchema: {
    type: 'object',
    properties: {
      expression: { type: 'string' },
    },
    required: ['expression'],
  },
  execute: async ({ input }) => {
    try {
      const result = eval(input.expression);
      return { isError: false, data: result.toString() };
    } catch (error) {
      return { isError: true, data: error.message };
    }
  },
});

const agent = new Agent({
  name: 'math-buddy',
  provider: new OllamaProvider({ model: 'llama2' }),
  tools: [calculatorTool],
  instructions: 'You are a helpful math assistant.',
});

const result = await agent.execute({
  messages: [
    {
      role: 'user',
      content: [{ type: 'text', text: 'What is 15 * 7?' }],
    },
  ],
});
```

### Weather agent

Create an agent that provides weather information:

```typescript
import { Agent } from '@agenite/agent';
import { Tool } from '@agenite/tool';
import { OpenAIProvider } from '@agenite/openai';

const weatherTool = new Tool({
  name: 'weather',
  description: 'Get current weather for a location',
  version: '1.0.0',
  inputSchema: {
    type: 'object',
    properties: {
      location: { type: 'string' },
      units: { 
        type: 'string',
        enum: ['metric', 'imperial'],
        default: 'metric',
      },
    },
    required: ['location'],
  },
  execute: async ({ input }) => {
    try {
      const weather = await getWeather(input.location, input.units);
      return {
        isError: false,
        data: JSON.stringify(weather),
      };
    } catch (error) {
      return {
        isError: true,
        data: 'Failed to get weather data',
      };
    }
  },
});

const agent = new Agent({
  name: 'weather-assistant',
  provider: new OpenAIProvider(),
  tools: [weatherTool],
  instructions: 'You help users get weather information.',
});
```

## Intermediate examples

### Stateful todo manager

Create an agent that manages a todo list:

```typescript
interface TodoState {
  todos: Array<{
    id: string;
    text: string;
    completed: boolean;
  }>;
}

const todoTool = new Tool({
  name: 'todo',
  description: 'Manage todos',
  version: '1.0.0',
  inputSchema: {
    type: 'object',
    properties: {
      action: {
        type: 'string',
        enum: ['add', 'remove', 'toggle', 'list'],
      },
      id: { type: 'string' },
      text: { type: 'string' },
    },
    required: ['action'],
  },
  execute: async ({ input, context }) => {
    const todos = context.state.todos;
    
    switch (input.action) {
      case 'add':
        todos.push({
          id: Date.now().toString(),
          text: input.text,
          completed: false,
        });
        break;
      case 'remove':
        const index = todos.findIndex(t => t.id === input.id);
        if (index >= 0) todos.splice(index, 1);
        break;
      case 'toggle':
        const todo = todos.find(t => t.id === input.id);
        if (todo) todo.completed = !todo.completed;
        break;
    }
    
    return {
      isError: false,
      data: JSON.stringify(todos),
    };
  },
});

const agent = new Agent<TodoState>({
  name: 'todo-manager',
  provider: new OpenAIProvider(),
  tools: [todoTool],
  initialState: {
    todos: [],
  },
  instructions: 'You help users manage their todo list.',
});
```

### Streaming file processor

Create an agent that processes files with streaming:

```typescript
const fileProcessorTool = new Tool({
  name: 'file-processor',
  description: 'Process files',
  version: '1.0.0',
  inputSchema: {
    type: 'object',
    properties: {
      filePath: { type: 'string' },
      operation: {
        type: 'string',
        enum: ['analyze', 'transform'],
      },
    },
    required: ['filePath', 'operation'],
  },
  execute: async ({ input }) => {
    const stream = createReadStream(input.filePath);
    let result = '';
    
    for await (const chunk of stream) {
      result += await processChunk(chunk, input.operation);
    }
    
    return {
      isError: false,
      data: result,
    };
  },
});

const agent = new Agent({
  name: 'file-processor',
  provider: new OpenAIProvider(),
  tools: [fileProcessorTool],
});

const iterator = agent.iterate({
  messages: [
    {
      role: 'user',
      content: [
        {
          type: 'text',
          text: 'Process data.csv and analyze its contents',
        },
      ],
    },
  ],
  stream: true,
});

for await (const chunk of iterator) {
  console.log(chunk);
}
```

## Advanced examples

### Multi-agent data pipeline

Create a pipeline of specialized agents:

```typescript
const extractorAgent = new Agent({
  name: 'extractor',
  provider: new OpenAIProvider(),
  tools: [extractionTool],
  instructions: 'Extract structured data from raw input.',
});

const transformerAgent = new Agent({
  name: 'transformer',
  provider: new OpenAIProvider(),
  tools: [transformationTool],
  instructions: 'Transform extracted data into required format.',
});

const loaderAgent = new Agent({
  name: 'loader',
  provider: new OpenAIProvider(),
  tools: [loadingTool],
  instructions: 'Load transformed data into target system.',
});

const pipelineAgent = new Agent({
  name: 'pipeline',
  provider: new OpenAIProvider(),
  agents: [extractorAgent, transformerAgent, loaderAgent],
  instructions: 'Coordinate ETL pipeline execution.',
  execute: async function* (params) {
    const extracted = yield* extractorAgent.execute(params);
    const transformed = yield* transformerAgent.execute(extracted);
    return yield* loaderAgent.execute(transformed);
  },
});
```

### Collaborative research agents

Create a team of agents that work together:

```typescript
const researcherAgent = new Agent({
  name: 'researcher',
  provider: new OpenAIProvider(),
  tools: [searchTool, browserTool],
  instructions: 'Research topics and gather information.',
});

const analyzerAgent = new Agent({
  name: 'analyzer',
  provider: new OpenAIProvider(),
  tools: [nlpTool, statsTool],
  instructions: 'Analyze research findings.',
});

const writerAgent = new Agent({
  name: 'writer',
  provider: new OpenAIProvider(),
  tools: [writingTool],
  instructions: 'Write reports based on analysis.',
});

const researchTeam = new Agent({
  name: 'research-team',
  provider: new OpenAIProvider(),
  agents: [researcherAgent, analyzerAgent, writerAgent],
  instructions: 'Coordinate research team activities.',
  execute: async function* (params) {
    // Parallel research tasks
    const researchResults = await Promise.all(
      params.topics.map(topic =>
        researcherAgent.execute({ topic })
      )
    );
    
    // Sequential analysis and writing
    const analysis = yield* analyzerAgent.execute({
      research: researchResults,
    });
    
    return yield* writerAgent.execute({
      analysis,
      format: params.format,
    });
  },
});
```
